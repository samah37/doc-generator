const mongoose = require('../db/mongoose');
const Schema = mongoose.Schema;
/**
 * Shema de la base de données des utilisateurs (users)
 *@prop {string} last_name - LE nom de l'étudiant.
 *@prop {string} first_name -  LE prénom de l'étudiant.
 *@prop {string} signal_object -  L'object du mail (le prbléme retrouvé).
 *@prop {string} signal_problem: - Description du probléme.

 */

let Signal = new Schema({
    student_last_name: {
        type: String
    },
    student_first_name: {
        type: String
    },
    signal_object: {
        type: String
    },
    signal_problem: {
        type: String
    },
   
});


module.exports = mongoose.model('Signal', Signal);
