const mongoose = require('../db/mongoose');
const Schema = mongoose.Schema;
/**
 * Shema de la base de données des utilisateurs (users)
 *@prop {string} last_name - LE nom de l'étudiant.
 *@prop {string} first_name -  LE prénom de l'étudiant.
 *@prop {string} level -  LE niveau d'étude de l'étudiant.
 *@prop {string} group -  LE droupe l'étudiant.
 *@prop {string} matricule -  La matricurle de l'étudiant.
 *@prop {string} section -  La section de l'étudiant.

 */
let Student = new Schema({
    last_name: {
        type: String,
        required : true
    },
    first_name: {
        type: String ,
        required : true

    },
    matricule: {
        type: String,
        required : true
    },
    section:{
        type: String
    },
    level: {
        type: String
    },
    group: {
        type: String
    },
});

module.exports = mongoose.model('Student', Student);
